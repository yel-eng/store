`).join(''); const inWishlist = state.wishlist.some(w => w.productId === main.id); const globalIdx = c * total + i; html += `
${dots}
${getText(cat.arName,cat.enName)}
${getText(main.name,main.enName)}
${cat.lenses.length} تدرجات · ${totalStock} قطعة
من ${getPrice(main)} ${getCurrencySymbol()}
`; } } track.innerHTML = html; // تحديث النقاط (6 نقاط فقط تمثل الفئات الأصلية) document.getElementById('sliderDots').innerHTML = COLORS_DATA.map((_, i) => `
`).join(''); let activeCatIdx = 0, currentPos = 0, isDragging = false, dragStartX = 0, dragStartPos = 0, animFrame = null, isAnimating = false; function getCenterOffset() { const vp = document.querySelector('.slider-viewport'); return (vp.offsetWidth - CARD_W) / 2; } // ✅ دالة snapToIndex المُصلحة - تعيد التعيين عند الوصول للحدود function snapToIndex(idx) { isAnimating = true; // إعادة تعيين الفهرس للنطاق الأصلي [0, total-1] const normalizedIdx = ((idx % total) + total) % total; activeCatIdx = normalizedIdx; const target = idx * ITEM_W; const centerOffset = getCenterOffset(); const pos = target - centerOffset; cancelAnimationFrame(animFrame); animateTo(pos, () => { isAnimating = false; // ✅ إعادة التعيين الفوري للموضع عند الوصول للنهاية/البداية لمنع المساحة السوداء if (idx >= total * 2) { const resetPos = normalizedIdx * ITEM_W - centerOffset; track.style.transition = 'none'; track.style.transform = `translateX(-${resetPos}px)`; void track.offsetWidth; // force reflow track.style.transition = ''; } else if (idx < 0) { const resetPos = (normalizedIdx + total) * ITEM_W - centerOffset; track.style.transition = 'none'; track.style.transform = `translateX(-${resetPos}px)`; void track.offsetWidth; track.style.transition = ''; } }); updateDots(normalizedIdx); } function animateTo(targetPos, callback) { const start = currentPos, diff = targetPos - start, duration = 500, startTime = performance.now(); function step(now) { const elapsed = now - startTime, progress = Math.min(elapsed / duration, 1); const ease = 1 - Math.pow(1 - progress, 4); currentPos = start + diff * ease; track.style.transform = `translateX(-${currentPos}px)`; if (progress < 1) animFrame = requestAnimationFrame(step); else if (callback) callback(); } animFrame = requestAnimationFrame(step); } function updateDots(idx) { const normalized = ((idx % total) + total) % total; document.querySelectorAll('#sliderDots .slider-dot').forEach((d, i) => { const active = i === normalized; d.classList.toggle('active', active); d.setAttribute('aria-selected', active); }); } // ✅ التهيئة الأولية: نبدأ من النسخة الوسطى (النسخة #2) لضمان سلاسة التمرير في الاتجاهين const startCopy = 2; currentPos = (startCopy * total) * ITEM_W - getCenterOffset(); track.style.transform = `translateX(-${currentPos}px)`; updateDots(0); let autoplayInterval = setInterval(() => { if (!isDragging && !isAnimating) { activeCatIdx++; snapToIndex(activeCatIdx); } }, 3500); function resetAutoplay() { clearInterval(autoplayInterval); autoplayInterval = setInterval(() => { if (!isDragging && !isAnimating) { activeCatIdx++; snapToIndex(activeCatIdx); } }, 3500); } document.getElementById('prevBtn').onclick = () => { activeCatIdx--; snapToIndex(activeCatIdx); resetAutoplay(); }; document.getElementById('nextBtn').onclick = () => { activeCatIdx++; snapToIndex(activeCatIdx); resetAutoplay(); }; document.getElementById('sliderDots').onclick = (e) => { const dot = e.target.closest('.slider-dot'); if (!dot) return; const idx = parseInt(dot.dataset.idx); // الانتقال للنسخة الوسطى + الفهرس المطلوب const targetGlobalIdx = startCopy * total + idx; activeCatIdx = idx; snapToIndex(targetGlobalIdx); resetAutoplay(); }; track.onclick = (e) => { const card = e.target.closest('.slider-card'); if (!card || isDragging) return; const catIdx = parseInt(card.dataset.catIdx); openLensModal(catIdx); }; const startDrag = (x) => { isDragging = true; dragStartX = x; dragStartPos = currentPos; cancelAnimationFrame(animFrame); isAnimating = false; track.style.transition = 'none'; }; const moveDrag = (x) => { if (!isDragging) return; const diff = x - dragStartX; currentPos = dragStartPos - diff; track.style.transform = `translateX(-${currentPos}px)`; }; const endDrag = () => { if (!isDragging) return; isDragging = false; const centerOffset = getCenterOffset(); const rawIdx = (currentPos + centerOffset) / ITEM_W; let targetIdx = Math.round(rawIdx); if (Math.abs(currentPos - dragStartPos) > CARD_W * 0.3) { targetIdx = currentPos < dragStartPos ? Math.ceil(rawIdx) : Math.floor(rawIdx); } activeCatIdx = ((targetIdx % total) + total) % total; snapToIndex(targetIdx); resetAutoplay(); }; track.addEventListener('touchstart', e => startDrag(e.touches[0].clientX), {passive:true}); track.addEventListener('touchmove', e => moveDrag(e.touches[0].clientX), {passive:true}); track.addEventListener('touchend', endDrag); track.addEventListener('mousedown', e => { e.preventDefault(); startDrag(e.clientX); }); document.addEventListener('mousemove', e => moveDrag(e.clientX)); document.addEventListener('mouseup', endDrag); }
